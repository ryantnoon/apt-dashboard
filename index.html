Connect Supabase

Connect a free Supabase project to sync your status and pipeline data across all your devices.

Setup Instructions (one time)
  1. Create a free project at supabase.com
  2. Open the SQL Editor and run this:
CREATE TABLE apt_tracking (
  building_name TEXT PRIMARY KEY,
  status TEXT DEFAULT 'active',
  stage TEXT DEFAULT 'none',
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE apt_tracking
  ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Allow all"
  ON apt_tracking FOR ALL
  USING (true) WITH CHECK (true);
